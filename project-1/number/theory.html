Different types of numbers using JavaScript code

  • Prime Numbers

    • Prime numbers are numbers that have only two factors: 1 and themselves.
    Example:-
    function isPrime(num) { if (num <= 1) return false; if (num <= 3) return true; if (num % 2 === 0 || num % 3 === 0) return false; for (let i = 5; i * i <= num; i += 6) { if (num % i === 0 || num % (i + 2) === 0) return false; } return true; } console.log(isPrime(11)); // Output: true console.log(isPrime(25)); // Output: false

  • Buzz Number

    • A number is a Buzz number if it ends with 7 or is divisible by 7.
    Example:-
    function isBuzz(num) { return num % 10 === 7 || num % 7 === 0; } console.log(isBuzz(17)); // Output: true console.log(isBuzz(21)); // Output: true console.log(isBuzz(25)); // Output: false

  • Perfect Number

    • A perfect number is a positive integer that is equal to the sum of its proper divisors, excluding itself.
    Example:-
    function isPerfect(num) { let sum = 0; for (let i = 1; i <= num / 2; i++) { if (num % i === 0) { sum += i; } } return sum === num && num !== 0; } console.log(isPerfect(6)); // Output: true

  • Composite Number

    • A composite number is a positive integer that has more than two divisors.
    Example:-
    function isComposite(num) { if (num <= 1) return false; for (let i = 2; i <= Math.sqrt(num); i++) { if (num % i === 0) return true; } return false; } console.log(isComposite(4)); // Output: true console.log(isComposite(7)); // Output: false

  • Armstrong Number

    • An Armstrong number is a number that is equal to the sum of its own digits each raised to the power of the number of digits.
    Example:-
    function isArmstrong(num) { const digits = num.toString().length; const sum = num.toString().split('').reduce((acc, digit) => acc + Math.pow(parseInt(digit), digits), 0); return sum === num; } console.log(isArmstrong(153)); // Output: true console.log(isArmstrong(123)); // Output: false

  • Harshad Number

    • A Harshad number is a number that is divisible by the sum of its digits.
    Example:-
    function isHarshad(num) { const sum = num.toString().split('').reduce((acc, digit) => acc + parseInt(digit), 0); return num % sum === 0; } console.log(isHarshad(18)); // Output: true console.log(isHarshad(20)); // Output: false

  • Sad Number

    • A sad number is simply a number that is not a happy number.
    Example:-
    function isSad(num) { return !isHappy(num); } console.log(isSad(14)); // Output: true console.log(isSad(19)); // Output: false

  • Happy Number

    • A happy number is a number which eventually reaches 1 when replaced by the sum of the square of each digit.
    Example:-
    function isHappy(num) { let slow = num, fast = num; do { slow = sumOfSquares(slow); fast = sumOfSquares(sumOfSquares(fast)); if (slow === 1) return true; } while (slow !== fast); return false; } function sumOfSquares(num) { return num.toString().split('').reduce((sum, digit) => sum + Math.pow(parseInt(digit), 2), 0); } console.log(isHappy(19)); // Output: true console.log(isHappy(2)); // Output: false

  • Fibonacci Number

    • A Fibonacci number is a number in the Fibonacci sequence.
    Example:-
    function isFibonacci(num) { let a = 0, b = 1, c; for (let i = 0; i <= num; i++) { c = a + b; a = b; b = c; } return b === num; } console.log(isFibonacci(5)); // Output: true console.log(isFibonacci(10)); // Output: false

  • Triangular Number

    • A triangular number is a number that can be represented in the form of a triangular dot pattern.
    Example:-
    function isTriangular(num) { let sum = 0; for (let i = 1; sum < num; i++) { sum += i; if (sum === num) return true; } return false; } console.log(isTriangular(6)); // Output: true console.log(isTriangular(10)); // Output: false

  • Square Number

    • A square number is a number that can be represented in the form of a square dot pattern.
    Example:-
    function isSquare(num) { return Math.sqrt(num) % 1 === 0; } console.log(isSquare(4)); // Output: true console.log(isSquare(9)); // Output: true console.log(isSquare(10)); // Output: false

  • Cube Number

    • A cube number is a number that can be represented in the form of a cube dot pattern.
    Example:-
    function isCube(num) { return Math.cbrt(num) % 1 === 0; } console.log(isCube(8)); // Output: true console.log(isCube(27)); // Output: true console.log(isCube(28)); // Output: false

  • Kaprekar Number

    • A Kaprekar number is a number that is equal to the sum of its digits when squared and then split into two parts.
    Example:-
    function isKaprekar(num) { const square = num * num; const strSquare = square.toString(); for (let i = 1; i < strSquare.length; i++) { const left = parseInt(strSquare.slice(0, i)); const right = parseInt(strSquare.slice(i)); if (left + right === num && right !== 0) return true; } return false; } console.log(isKaprekar(5)); // Output: true console.log(isKaprekar(297)); // Output: true console.log(isKaprekar(100)); // Output: false
